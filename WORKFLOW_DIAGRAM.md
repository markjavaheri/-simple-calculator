```bash
git pull origin main

# Create new branch
git checkout -b feature/my-feature

# See what changed
git status

# Stage changes
git add .

# Commit changes
git commit -m "Description"

# Push to GitHub
git push origin feature/my-feature

# Switch branches
git checkout main
git checkout feature/my-feature
```
